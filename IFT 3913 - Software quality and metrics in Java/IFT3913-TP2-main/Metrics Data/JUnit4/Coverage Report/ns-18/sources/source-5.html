 package org.jfree.chart.text;
 
 /**
  * Used to indicate the position of an anchor point for a text block.
  */
 public enum TextBlockAnchor {
 
     /** Top/left. */
     TOP_LEFT,
 
     /** Top/center. */
     TOP_CENTER,
 
     /** Top/right. */
     TOP_RIGHT,
 
     /** Middle/left. */
     CENTER_LEFT,
 
     /** Middle/center. */
     CENTER,
 
     /** Middle/right. */
     CENTER_RIGHT,
 
     /** Bottom/left. */
     BOTTOM_LEFT,
 
     /** Bottom/center. */
     BOTTOM_CENTER,
 
     /** Bottom/right. */
     BOTTOM_RIGHT
 }
