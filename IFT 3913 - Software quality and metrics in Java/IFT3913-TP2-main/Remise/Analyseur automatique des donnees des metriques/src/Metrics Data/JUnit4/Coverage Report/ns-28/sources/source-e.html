 /**
  * Used to indicate one of three positions in a time period:
  * {@code START}, {@code MIDDLE} and {@code END}.
  */
 public enum TimePeriodAnchor {
 
     /** Start of period. */
     START,
 
     /** Middle of period. */
     MIDDLE,
 
     /** End of period. */
     END
 
 }
